Is a number prime? (JavaScript, 6 kyu) 

    function isPrime(num) {
        let isPrimeNumber = true; 
        
        if(num <= 1) return false;
        
        for (let i = 2; i <= Math.sqrt(num); i++) {
            if (num % i === 0) isPrimeNumber = false; 
        } 
        
        return isPrimeNumber; 
    }
